(define (make-segment p q)
  (cons p q))

(define (start-segment s)
  (car s))

(define (end-segment s)
  (cdr s))

(define (make-point x y)
  (cons x y))

(define (x-point p)
  (car p))

(define (y-point p)
  (cdr p))

(define (midpoint-segment s)
  (let ((p (start-segment s))
        (q (end-segment s)))
    (make-point (average (x-point p)
                         (x-point q))
                (average (y-point p)
                         (y-point q)))))
\endtt
